☢ Write a program add Natural numbers :

Program: 
#include <stdio.h>
int main()
{
    int n, k = 0;
    printf("Enter the number \n");
    scanf("%d", &n);
    for (int i = 1; i <= n; i++)
    {
        k = k + i;
    }
    printf("Sum of %d Natural number is = %d", n, k);
    return 0;
}

☢ Write a program to calculate leap year

Program: 
#include <stdio.h>
int main()
{
    int year;
    printf("Enter the number \n");
    scanf("%d", &year);
    if (year % 4 == 0 && year % 100 != 0 || year % 400 == 0)
    {
        printf("This is leap year\n");
    }
    else
    {
        printf("This is not leap year\n");
    }

    return 0;
}

☢ Write a program to print exponential of a number

Program: 
#include <stdio.h>
int main()
{
    int num, power, k = 1;
    printf("Enter the number and power\n");
    scanf("%d %d", &num, &power);
    for (int i = 1; i <= power; i++)
    {
        k = k * num;
    }

    printf("%d", k);
    return 0;
}

☢ Write a program to print reverse a number

Program: 
#include <stdio.h>
int main()
{
    int num, k;
    printf("Enter the number \n")
    scanf("%d", &num);
    for (int i = 0; num != 0; i++)
    {
        k = num % 10;
        num = num / 10;
        printf("%d", k);
    }

    return 0;
}
