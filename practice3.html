☢ Write a program if the given number is palindrome or not :

Program: 
#include <stdio.h>
int main()
{
    int num, k, temp, sum = 0;
    scanf("%d", &num);
    temp = num;
    for (int i = 0; num != 0; i++)
    {
        k = num % 10;
        num = num / 10;
        sum = (sum * 10) + k;
    }
    if (temp == sum)
    {
        printf("This number is palindrome\n");
    }
    else
    {
        printf("This number is not palindrome\n");
    }

    return 0;

    {
        printf("This is not prime number\n");
    }

    {
        printf("This is prime number\n");

    {
        if (num % i == 0)
